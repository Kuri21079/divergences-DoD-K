		exists = c:FRA
		root = c:FRA

		NOT = { has_variable = fra_dealing_with_occ }
	}

		has_technology_researched = nationalism
	}

			type = je_fra_annex_occ
		}

		ig:ig_landowners = {
			add_modifier = {
				name = ig_manifest_destiny
				months = normal_modifier_time
			}
		}
		ig:ig_intelligentsia = {
			add_modifier = {
				name = ig_manifest_destiny
				months = normal_modifier_time
				multiplier = -1
			}
		}
		# Add Claims
		s:STATE_LANGUEDOC = { 
			add_claim = ROOT
		}
		s:STATE_RHONE = { 
			add_claim = ROOT
		}
		s:STATE_ARMAGNAC = { 
			add_claim = ROOT
		}
		s:STATE_AUVERGNE = { 
			add_claim = ROOT
		}
		s:STATE_AQUITAINE = { 
			add_claim = ROOT
		}
		set_variable = fra_dealing_with_occ
    }

	ai_chance = {
		base = 100
	}
}
