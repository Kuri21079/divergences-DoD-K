
dvg_edward_statue = {
	is_shown = {
	
	        exists = c:LOT

		this = c:LOT
		
		NOT = { has_global_variable = dvg_edward_statue_built}
	}

    possible = {
	OR = {
		has_law = law_type:law_autocracy
		has_law = law_type:law_monarchy
		}
	
	}

    when_taken = {
		set_global_variable = dvg_edward_statue_built
		c:BEO = {
			change_relations = { 
				country = ROOT
				value = 15
			}
		   }

	
}
ai_chance = {
		base = 100
	}
}

			set_state_owner = c:TGI
		}

	}
	

	
}
